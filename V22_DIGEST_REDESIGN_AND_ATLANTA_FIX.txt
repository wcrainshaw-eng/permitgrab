PART A — FIX: MAKE THE DIGEST ACTUALLY SEND
================================================================================

ROOT CAUSE:

      User.query.filter(User.digest_active == True, ...)
  - The `User` model maps to a `users` table that was NEVER created in SQLite.
  - Result: 0 eligible users found → 0 emails sent → blank digest every day.
  - Meanwhile, actual subscriber data is in `/var/data/subscribers.json`:

FIX — Option chosen: Read from subscribers.json (quickest path to working digest)

Do NOT try to create a `users` table and migrate data. That's a bigger project.
Instead, rewrite `send_daily_digest()` to read from subscribers.json directly.

Step 1: Add a function to load subscribers from JSON

In email_alerts.py, add near the top (after imports):

  import json
  from pathlib import Path

  SUBSCRIBERS_FILE = Path("/var/data/subscribers.json")

  def load_subscribers():
      """Load active subscribers from subscribers.json."""
      if not SUBSCRIBERS_FILE.exists():
          print(f"  [WARN] Subscribers file not found: {SUBSCRIBERS_FILE}")
          return []
      try:
          with open(SUBSCRIBERS_FILE) as f:
              subscribers = json.load(f)
          return [s for s in subscribers if s.get("active", False)]
      except Exception as e:
          print(f"  [ERROR] Failed to load subscribers: {e}")
          return []


Step 2: Create a lightweight subscriber wrapper

Create a simple wrapper class:

  class SubscriberProxy:
      """Adapter that makes a subscribers.json dict look like a User model."""
      def __init__(self, data):
          self.email = data.get("email", "")
          self.name = data.get("name", "")
          self.plan = data.get("plan", "free")
          self.unsubscribe_token = data.get("unsubscribe_token", "")
          self.digest_active = data.get("active", True)

          # Convert single city string to JSON array format

          cities = data.get("cities", [])  # Support future multi-city
          if cities:
              self.digest_cities = json.dumps(cities)
          elif city:
              self.digest_cities = json.dumps([city])
          else:
              self.digest_cities = "[]"

Step 2: Add a distance helper to city_configs.py

  import math

  def haversine_miles(lat1, lon1, lat2, lon2):
      """Calculate distance in miles between two lat/lon points."""
      R = 3959  # Earth radius in miles
      dlat = math.radians(lat2 - lat1)
      dlon = math.radians(lon2 - lon1)
      a = (math.sin(dlat/2)**2 +
           math.cos(math.radians(lat1)) * math.cos(math.radians(lat2)) *
           math.sin(dlon/2)**2)
      return R * 2 * math.asin(math.sqrt(a))


  def get_nearby_cities(city_key, max_miles=150, limit=5):
      """Find active cities within max_miles of the given city.
      Returns list of (city_key, config, distance) sorted by distance."""
      config = CITY_REGISTRY.get(city_key)
      if not config or not config.get("lat") or not config.get("lon"):
          return []

      lat1, lon1 = config["lat"], config["lon"]
      nearby = []

      for key, cfg in CITY_REGISTRY.items():
          if key == city_key:
              continue
          if not cfg.get("active") or not cfg.get("lat") or not cfg.get("lon"):
              continue
          dist = haversine_miles(lat1, lon1, cfg["lat"], cfg["lon"])
          if dist <= max_miles:
              nearby.append((key, cfg, dist))

      nearby.sort(key=lambda x: x[2])
      return nearby[:limit]

  def get_nearby_permits(subscriber_city, since_date, limit=3):
      """Get a few permits from the nearest city that has fresh data.
      V22: Uses lat/lon from CITY_REGISTRY + haversine distance."""
      from city_configs import get_nearby_cities, CITY_REGISTRY

      nearby_cities = get_nearby_cities(city_key, max_miles=150, limit=10)
      if not nearby_cities:

      # Try each nearby city (closest first) until one has recent permits
      conn = permitdb.get_connection()
      for nearby_key, cfg, dist in nearby_cities:
          nearby_city_name = cfg["name"]
          nearby_state = cfg.get("state", "")

          query = """
              SELECT * FROM permits
              WHERE city = ? AND state = ?
              AND filing_date >= ?
              ORDER BY filing_date DESC, estimated_cost DESC
              LIMIT ?
          """
          cursor = conn.execute(query, [nearby_city_name, nearby_state, since_date, limit])
          permits = [dict(row) for row in cursor]
          if permits:
              return permits, nearby_city_name, nearby_state

      return [], None, None


  def _get_same_state_permits(subscriber_city, since_date, limit=3):
      """Last-resort fallback: find permits from any other city in same state."""
      conn = permitdb.get_connection()
      state_query = "SELECT state FROM permits WHERE city = ? LIMIT 1"
      row = conn.execute(state_query, [subscriber_city.title()]).fetchone()
      if not row:
          return [], None, None
      state = row[0]

      query = """
          SELECT * FROM permits
          WHERE state = ? AND city != ?
          AND filing_date >= ?
          ORDER BY filing_date DESC, estimated_cost DESC
          LIMIT ?
      """
      cursor = conn.execute(query, [state, subscriber_city.title(), since_date, limit])
      permits = [dict(row) for row in cursor]
      if permits:
          nearby_city = permits[0].get("city", "Nearby")
          return permits, nearby_city, state
      return [], None, None

  def get_market_snapshot(days=7):
      """Get aggregate stats for the market snapshot section.
      V22: Simple stats across all cities for the past week."""
      conn = permitdb.get_connection()
      since = (datetime.now() - timedelta(days=days)).strftime('%Y-%m-%d')

      stats = {}

      # Total permits this week
      row = conn.execute(
          "SELECT COUNT(*) as cnt FROM permits WHERE filing_date >= ?",
          [since]
      ).fetchone()
      stats["total_permits"] = row[0] if row else 0

      # Total value
      row = conn.execute(
          "SELECT SUM(estimated_cost) as total FROM permits WHERE filing_date >= ? AND estimated_cost > 0",
          [since]
      ).fetchone()
      stats["total_value"] = row[0] if row else 0

      # Top trade category
      row = conn.execute("""
          SELECT trade_category, COUNT(*) as cnt
          FROM permits WHERE filing_date >= ? AND trade_category IS NOT NULL AND trade_category != ''
          GROUP BY trade_category ORDER BY cnt DESC LIMIT 1
      """, [since]).fetchone()
      stats["top_trade"] = row[0] if row else "General"
      stats["top_trade_count"] = row[1] if row else 0

      # Busiest city
      row = conn.execute("""
          SELECT city, state, COUNT(*) as cnt
          FROM permits WHERE filing_date >= ?
          GROUP BY city, state ORDER BY cnt DESC LIMIT 1
      """, [since]).fetchone()
      stats["busiest_city"] = row[0] if row else ""
      stats["busiest_state"] = row[1] if row else ""
      stats["busiest_count"] = row[2] if row else 0

      return stats
