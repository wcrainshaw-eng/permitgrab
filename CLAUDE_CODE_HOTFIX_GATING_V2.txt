FIX:
The score must be calculated from ACTUAL permit attributes. Here's a formula
that will produce real differentiation:

def calculate_lead_score(permit):
    score = 50  # base score

    # Value component (0-20 points)
    value = permit.get('valuation', 0) or permit.get('estimated_cost', 0) or 0
    if isinstance(value, str):
        value = float(value.replace('$','').replace(',','')) if value else 0
    if value >= 1000000: score += 20
    elif value >= 500000: score += 16
    elif value >= 200000: score += 12
    elif value >= 100000: score += 8
    elif value >= 50000: score += 4

    # Recency component (0-15 points)
    from datetime import datetime, timedelta
    date_str = permit.get('issue_date') or permit.get('filed_date') or ''
    try:
        permit_date = datetime.strptime(str(date_str)[:10], '%Y-%m-%d')
        days_old = (datetime.now() - permit_date).days
        if days_old <= 3: score += 15
        elif days_old <= 7: score += 12
        elif days_old <= 14: score += 8
        elif days_old <= 30: score += 4
    except: pass

    # Contact info completeness (0-10 points)
    if permit.get('phone') or permit.get('contact_phone'): score += 5
    if permit.get('contact_name') or permit.get('owner_name'): score += 5

    # Permit type (0-5 points)
    ptype = str(permit.get('permit_type', '') or permit.get('work_type', '')).lower()
    if 'new' in ptype or 'construction' in ptype: score += 5
    elif 'addition' in ptype or 'renovation' in ptype: score += 3
    elif 'alteration' in ptype or 'repair' in ptype: score += 1

    # Cap at 99
    return min(max(score, 40), 99)

This formula guarantees:
- A $2M new construction permit filed today with contact info = 95+
- A $50K repair permit from 3 weeks ago with no contact info = 55
- Spread of at least 30-40 points across a typical city's permits
