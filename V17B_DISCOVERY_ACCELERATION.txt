  def run_city_discovery_batch(batch_size=100):
      """
      Check the next batch of cities by population that haven't
      been searched yet. Updates us_cities.status as it goes
      so it picks up where it left off next cycle.
      """
      conn = permitdb.get_db()

      # Get next batch of unsearched cities, largest first
      cities = conn.execute('''
          SELECT id, city_name, state, county, population, slug
          FROM us_cities
          WHERE status = 'not_started'
          ORDER BY population DESC
          LIMIT ?
      ''', (batch_size,)).fetchall()

      if not cities:
          print("[Discovery] All cities have been searched!")
          return

      print(f"[Discovery] Checking {len(cities)} cities "
            f"(pop {cities[0]['population']:,} to {cities[-1]['population']:,})")

      discovered = 0
      for city in cities:
          city_name = city['city_name']
          state = city['state']

          # Mark as searching so we don't re-check
          conn.execute('''
              UPDATE us_cities SET status='searching',
              last_searched_at=datetime('now'),
              search_attempts = search_attempts + 1
              WHERE id = ?
          ''', (city['id'],))
          conn.commit()

          found = False

          # Check 1: Socrata — search for "{city} building permits"
          try:
              results = search_socrata_catalog(
                  f"{city_name} building permits",
                  domain=None  # Search all domains
              )
              for r in results:
                  if test_socrata_endpoint(r['domain'], r['dataset_id']):
                      add_discovered_source(city_name, state, r)
                      found = True
                      discovered += 1
                      break
          except Exception as e:
              pass

          # Check 2: ArcGIS Hub — search for city permit services
          if not found:
              try:
                  results = search_arcgis_hub(
                      f"{city_name} {state} building permits"
                  )
                  for r in results:
                      if test_arcgis_endpoint(r['url']):
                          add_discovered_source(city_name, state, r)
                          found = True
                          discovered += 1
                          break
              except Exception as e:
                  pass

          # Update us_cities status
          if found:
              conn.execute('''
                  UPDATE us_cities SET status='has_source'
                  WHERE id = ?
              ''', (city['id'],))
          else:
              conn.execute('''
                  UPDATE us_cities SET status='no_data_available'
                  WHERE id = ?
              ''', (city['id'],))
          conn.commit()

          # Rate limit: 2 seconds between cities
          time.sleep(2)

      print(f"[Discovery] Batch complete. Found {discovered}/{len(cities)}")

The state discovery only checks Socrata. Many cities use ArcGIS.

  def search_arcgis_hub(query):
      """
      Search ArcGIS Hub for open data FeatureServers.
      Free API, no key needed.
      """
      url = "https://hub.arcgis.com/api/feed/dcat-us/1.1"
      # Alternative: https://opendata.arcgis.com/api/v3/search
      params = {
          'q': query,
          'per_page': 10
      }
      try:
          resp = requests.get(
              "https://opendata.arcgis.com/api/v3/search",
              params={'q': query, 'per_page': 10},
              timeout=15
          )
          if resp.ok:
              data = resp.json()
              results = []
              for item in data.get('results', []):
                  # Look for FeatureServer URLs
                  for dist in item.get('distributions', []):
                      url = dist.get('accessURL', '')
                      if 'FeatureServer' in url and 'permit' in item.get('title','').lower():
                          results.append({
                              'title': item['title'],
                              'url': url,
                              'platform': 'arcgis',
                              'source': 'arcgis_hub'
                          })
              return results
      except Exception:
          pass
      return []

  def test_arcgis_endpoint(url):
      """Test if an ArcGIS FeatureServer has permit data."""
      query_url = f"{url}/query" if not url.endswith('/query') else url
      params = {
          'where': '1=1',
          'outFields': '*',
          'resultRecordCount': 5,
          'f': 'json'
      }
      try:
          resp = requests.get(query_url, params=params, timeout=15)
          if resp.ok:
              data = resp.json()
              features = data.get('features', [])
              if len(features) > 0:
                  return True
      except Exception:
          pass
      return False
