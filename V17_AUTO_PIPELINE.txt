      try:

          print("Starting daily source discovery...")
          run_state_discovery()    # Check state portals
          run_county_discovery()   # Check county portals (if implemented)
          mark_daily_complete('discovery')
      except Exception as e:
          print(f"Discovery error: {e}")

Implement should_run_daily() and mark_daily_complete() using a simple
key-value store in SQLite:

  CREATE TABLE IF NOT EXISTS system_state (
      key TEXT PRIMARY KEY,
      value TEXT,
      updated_at TEXT DEFAULT (datetime('now'))
  );

  should_run_daily('discovery'):
    SELECT value FROM system_state WHERE key='last_discovery_run'
    → If NULL or >24 hours ago, return True

  mark_daily_complete('discovery'):
    INSERT OR REPLACE INTO system_state (key, value)
    VALUES ('last_discovery_run', datetime('now'))

This way discovery runs daily on the same Render process —
no separate cron job, no manual kicking off on the shell.

STEP 2: AUTO-TEST PENDING CITIES (20 min)
===================================================================

After discovery runs, there may be new 'pending' entries in prod_cities.
Build a function that auto-tests them and activates the good ones.

New function in collector.py:

  def activate_pending_cities():
      """
      Test all pending prod_cities entries.
      If we can successfully collect permits, flip to active.
      If endpoint is dead/empty after 3 attempts, flip to failed.
      """
      pending = permitdb.get_prod_cities(status='pending')
      activated = []
      failed = []

      for city in pending:
          source_id = city['source_id']
          config = _get_source_config(source_id)

          if not config:
              # No config found — try to build one from prod_cities metadata
              # (source_endpoint, source_type fields)
              config = _build_config_from_prod_city(city)

          if not config:
              failed.append(city['slug'])
              continue

          try:
              # Try collecting last 30 days
              if config.get('mode') == 'bulk':
                  results = collect_bulk_source(source_id, config, days_back=30)
                  permit_count = sum(len(v) for v in results.values())
              else:
                  permits = fetch_permits(source_id, days_back=30)
                  permit_count = len(permits) if permits else 0

              if permit_count > 0:
                  # SUCCESS — activate this city
                  permitdb.update_prod_city_status(
                      city['slug'], 'active',
                      notes=f'Auto-activated: {permit_count} permits found'
                  )
                  activated.append({
                      'city': city['name'],
                      'state': city['state'],
                      'slug': city['slug'],
                      'permits': permit_count
                  })
              else:
                  # No permits but endpoint worked — increment attempts
                  attempts = city.get('consecutive_failures', 0) + 1
                  if attempts >= 3:
                      permitdb.update_prod_city_status(
                          city['slug'], 'failed',
                          notes='Auto-failed: 3 attempts, 0 permits'
                      )
                      failed.append(city['slug'])
                  else:
                      # Leave as pending, try again tomorrow
                      permitdb.increment_failure(city['slug'])

          except Exception as e:
              attempts = city.get('consecutive_failures', 0) + 1
              if attempts >= 3:
                  permitdb.update_prod_city_status(
                      city['slug'], 'failed',
                      notes=f'Auto-failed: {str(e)[:200]}'
                  )
                  failed.append(city['slug'])

      return activated, failed

Call this right after discovery in the daily schedule:

  if should_run_daily('discovery'):

      run_county_discovery()
      activated, failed = activate_pending_cities()

New function in email_alerts.py:

  def send_new_cities_alert(activated_cities):
      """
      Send email to owner when new cities are auto-activated.
      This is the trigger for SEO content creation.
      """
      if not activated_cities:
          return

      subject = f"PermitGrab: {len(activated_cities)} New Cities Activated"

      # Build the email body
      city_rows = []
      for city in activated_cities:
          city_rows.append(
              f"  • {city['city']}, {city['state']} — "
              f"{city['permits']} permits found — "
              f"https://permitgrab.com/permits/{city['slug']}"
          )

      body = f"""
New cities were automatically discovered and activated on PermitGrab.

NEWLY ACTIVATED ({len(activated_cities)} cities):
{chr(10).join(city_rows)}

SEO ACTION ITEMS:
  1. Review each city page for accuracy
  2. Write a city-specific blog post for high-population cities
  3. Add internal links from related state/trade pages
  4. Check that permits are displaying correctly
  5. Consider adding to Google Search Console for monitoring

These cities are now:
  ✓ Live on the site with auto-generated pages
  ✓ Included in sitemap.xml
  ✓ Set to index,follow (if permits > 0)
  ✓ Collecting permits every 6 hours
  ✓ Available for user subscriptions

No action needed for the pages themselves — they're already live.
The SEO content (blog posts, enhanced descriptions) is what adds value.
"""

      send_email(
          to='wcrainshaw@gmail.com',
          subject=subject,
          body=body
      )

      # Also log to system_state for the /api/health endpoint
      permitdb.set_system_state(
          'last_cities_activated',
          json.dumps({
              'date': datetime.now().isoformat(),
              'count': len(activated_cities),
              'cities': [c['slug'] for c in activated_cities]
          })
      )

STEP 5: BULK SOURCE AUTO-REGISTRATION (15 min)
===================================================================

When discovery finds a new county or state source, it currently
just adds one entry to prod_cities. But a county source covers
MANY cities. We need to:

  A) When a county/state source is discovered and tested:
     1. Add the source config to BULK_SOURCES (or a new
        bulk_sources table in SQLite)
     2. Run an initial collection from it
     3. Extract all unique cities from the city_field
     4. For EACH city found in the data:
        - Check if it exists in us_cities table
        - If yes: add to prod_cities with source_scope='county'
        - Update us_cities status to 'covered_by_county'

  B) This means one county discovery could add 20-50 cities
     to prod_cities in one shot.

  C) The email alert should group these:
     "Discovered: Cook County IL source → 130 new cities activated"

  D) Store the bulk source config in the database (not just
     city_configs.py) so discovery can add sources without
     modifying Python files:

     CREATE TABLE IF NOT EXISTS discovered_sources (
         id INTEGER PRIMARY KEY AUTOINCREMENT,
         source_key TEXT UNIQUE NOT NULL,
         name TEXT NOT NULL,
         state TEXT NOT NULL,
         platform TEXT NOT NULL,     -- socrata, arcgis, ckan
         mode TEXT DEFAULT 'bulk',
         endpoint TEXT NOT NULL,
         dataset_id TEXT,
         city_field TEXT,
         date_field TEXT,
         field_map TEXT,             -- JSON string
         scope TEXT,                 -- state, county, city
         discovered_at TEXT DEFAULT (datetime('now')),
         last_tested TEXT,
         status TEXT DEFAULT 'active',
         notes TEXT
     );

  The collector should check this table IN ADDITION to the
  hardcoded BULK_SOURCES dict. This way discovery can add
  new sources that immediately start collecting without
  any code changes or deploys.
